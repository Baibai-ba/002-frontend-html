# =========================
# 📝 Git 单人开发速查表
# =========================

# 🚀 初始化 / 克隆项目
git clone <仓库地址>

# 📌 查看状态 / 分支
git status              # 查看修改情况
git branch              # 查看本地分支
git branch -a           # 查看所有分支（含远程）

# ✍️ 提交代码三步曲
git add <文件名>        # 添加单个文件
git add .               # 添加所有修改
git commit -m "描述修改内容"
git push origin main    # 推送到远程 main

# 🔄 同步远程更新
git pull origin main

# 🌿 分支操作
git branch <分支名>            # 创建分支
git checkout <分支名>          # 切换分支
git push origin <分支名>       # 推送分支到远程

# 示例：创建备份分支
git branch backup-20250918
git push origin backup-20250918

# ⏪ 回退版本（危险操作⚠️）
git reset --hard <commit_id>
git push origin main --force

# 📖 小口诀
# 三部曲：改代码 → add → commit → push
# 两工具：status 看情况，checkout 换分支
# 一个保险：branch 建备份，随时能回滚
